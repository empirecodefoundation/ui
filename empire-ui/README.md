## Installation

```bash
npm install @empire-ui/core
# or
yarn add @empire-ui/core
# or
pnpm add @empire-ui/core
```

## Quick Start

```tsx
import { Button } from "@empire-ui/core";

function App() {
  return (
    <Button variant="primary" size="md">
      Click me
    </Button>
  );
}
```
